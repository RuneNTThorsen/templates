% Bibliography
\usepackage[backend=biber,
style=authoryear-comp,  % Citation style as (AUTHOR YEAR)

sortcites=true,
dashed=false,
maxcitenames=2,
maxbibnames=5,
uniquelist=false,

% Change bib-order
